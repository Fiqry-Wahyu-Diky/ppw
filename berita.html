datas = []

# ==== perulangan pages berita =====
for ipages in range(1,16):
  # ==== mengambil link setiap list berita ===
  for beritas in listberita:
    response_berita = requests.get(beritas+"indeks/"+str(ipages))

    namecategori = beritas.split("//")[1].split(".")
    print(f"... Mengambil data berita detik {namecategori[0]}, Pages ke-{ipages}")

    soup_berita = BeautifulSoup(response_berita.text, 'html.parser')
    pages_berita = soup_berita.findAll('article', {'class' :'list-content__item'})

    for items in pages_berita:
      cek_link_in = items.findAll("h3","media__title")
      if cek_link_in:
        link_in = items.find("h3","media__title")
      get_link_in = link_in.find("a").get("href")
      # print(get_link_in)
      # ======== masuk ke link pages ==========
      response_artikel = requests.get(get_link_in)
      soup_artikel = BeautifulSoup(response_artikel.text, 'html.parser')

      # =========== judul =========
      cek_judul = soup_artikel.findAll("h1","detail__title")
      if cek_judul:
        judul = soup_artikel.find("h1","detail__title").text.replace("\n","")
      else:
        judul=""
      # print("judul : ",judul)

      # ========== label ==========
      # cek_label = soup_artikel.find_all("span","detail__label")
      # # print(cek_label)
      # if cek_label:
      #   label = soup_artikel.find("span","detail__label").text.replace("Sepakbola","Sport").replace("\n","")
      # else:
      #   label = None
      label = namecategori[0]

      # ======= tempat ==========
      cek_location = soup_artikel.find_all("strong")
      if cek_location:
        location = soup_artikel.find("strong").text.replace("\n","")
      else:
        location = ""

      # =========== author =========
      cek_author = soup_artikel.find_all("div","detail__author")
      # print(cek_author)
      if cek_author:
        author = soup_artikel.find("div","detail__author").text.replace(label,"").replace(" -",'').replace("\n","")
      else:
        author = None
      # print(author)

      # ========== date =============
      cek_date = soup_artikel.findAll("div","detail__date")
      if cek_date:
        date = soup_artikel.find("div","detail__date").text.replace("\n","")
      else:
        date = ""
      # print(date)

      # ======= cek iklan ======
      cek_iklan = soup_artikel.find_all("div","parallaxindetail scrollpage")
      if cek_iklan:
        iklan = soup_artikel.find("div","parallaxindetail scrollpage").text.replace("\n","")


      # ====== cek lainnya ======
      cek_link_sisip = soup_artikel.find_all("table","linksisip")
      if cek_link_sisip:
        link_sisip = soup_artikel.find("table","linksisip").text.replace("\n","")
      else:
        link_sisip = ""

      cek_bodytag = soup_artikel.find_all("div","detail__body-tag mgt-16")
      if cek_bodytag:
        bodytag = soup_artikel.find("div","detail__body-tag mgt-16").text.replace("\n","")
      else:
        bodytag = ""

      cek_advertise = soup_artikel.find_all("div","ads-scrollpage-height")
      if cek_advertise:
        advertise = soup_artikel.find("div","ads-scrollpage-height").text
        # print(advertise)

      # ========== artikel ===========
      cek_artikel = soup_artikel.find_all("div","detail__body-text itp_bodycontent")
      if cek_artikel:
        arrArtikel = soup_artikel.findAll("p")
        artikel = ""
        for artikels in arrArtikel:
          text = artikels.get_text().replace(iklan,"").replace(link_sisip,"").replace(bodytag,"").replace(location,"").replace("\n","").replace(" -",'').replace("\r\t\t\t\t\t\t\t","").replace(advertise,"").replace("\n\rADVERTISEMENT\r\n\rSCROLL TO CONTINUE WITH CONTENT\r\n","")
          artikel += text + "\n"
      else:
        artikel = None
      # print(artikel)

      datas.append({
        'Tanggal' : date,
        'Penulis':author,
        "Tempat" : location,
        'Judul': judul,
        'Artikel': artikel,
        'Label' :  label
            })
